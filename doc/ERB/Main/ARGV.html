def ARGV.req_arg
  (@maybe_arg || self.shift || raise('missing argument')).tap {
    @maybe_arg = nil
  }
end

def ARGV.switch
  return nil if self.empty?
  arg = self.shift
  return nil if arg == '--'
  case arg
  when /\A-(.)(.*)/
    if $1 == '-'
      arg, @maybe_arg = arg.split(/=/, 2)
      return arg
    end
    raise 'unknown switch "-"' if $2[0] == ?- and $1 != 'T'
    if $2.size > 0
      self.unshift "-#{$2}"
      @maybe_arg = $2
    else
      @maybe_arg = nil
    end
    "-#{$1}"
  when /\A(\w+)=/
    arg
  else
    self.unshift arg
    nil
  end
end
